'use client';

import { useState } from 'react';

export default function Home() {
  const [messages, setMessages] = useState<string[]>([]);
  const [input, setInput] = useState('');
  const [loading, setLoading] = useState(false);

  const sendMessage = () => {
    if (!input.trim()) return;
    setLoading(true);

    // Simulación de respuesta IA (reemplazá con llamada real)
    setTimeout(() => {
      setMessages(prev => [...prev, input, "Respuesta de Bisnes IA"]);
      setInput('');
      setLoading(false);
    }, 1000);
  };

  return (
    <main className="p-4">
      <h1 className="text-2xl font-bold mb-4">Bienvenido a Bisnes IA</h1>

      <div className="mb-4">
        {messages.map((msg, idx) => (
          <p key={idx}>{msg}</p>
        ))}
      </div>

      <div className="flex gap-2">
        <input
          type="text"
          value={input}
          onChange={e => setInput(e.target.value)}
          placeholder="Preguntale algo a Bisnes IA..."
          className="border p-2 flex-grow"
          disabled={loading}
        />
        <button
          onClick={sendMessage}
          disabled={loading}
          className="bg-black text-white px-4 py-2 rounded"
        >
          Enviar
        </button>
      </div>
    </main>
  );
}
